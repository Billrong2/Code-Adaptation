	public void setDatesInMonth(ArrayList<DateTime> datesInMonth) {

	}

	
	public InfiniteViewPager(Context context, AttributeSet attrs) {
		super(context, attrs);
	}

	public InfiniteViewPager(Context context) {

	}

	@Override

		super.setAdapter(adapter);
